            var mouseX = 0;

            if (__is_ff) {  
                mouseX = e.layerX + document.body.scrollLeft;  
                mouseY = e.layerY + document.body.scrollLeft;

                mouseY = e.y + document.body.scrollTop;  
            }  
            return { x: mouseX, y: mouseY };
